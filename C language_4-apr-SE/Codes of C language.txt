code:-

	printf("%d > %d = %d\n",a,b, a>b);
	printf("%d < %d = %d\n",a,b, a<b);
	printf("%d >= %d = %d\n",a,b, a>=b);

4. Logical operators:-

code:-

#include<stdio.h>
main()
{
	int x,y;
	printf("Enter the value of x :-");
	scanf("%d",&x);
	printf("Enter the value of y :-");
	scanf("%d",&y);
	printf("The value of logical AND => %d\n",x>y && x<y);
	printf("The value of logical OR => %d\n",x>y || x<y);
	printf("The value of logical NOT => %d\n", !(x>y && x<y));
